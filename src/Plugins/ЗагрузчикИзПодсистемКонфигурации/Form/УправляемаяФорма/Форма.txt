	ПараметрыОткрытия = Новый Структура("ОтборПоИмениТеста", "");
	ИмяФормыВыбораПодсистемыУФ = СтрЗаменить(ЭтаФорма.ИмяФормы, "УправляемаяФорма", "ФормаВыбораПодсистемУФ");
